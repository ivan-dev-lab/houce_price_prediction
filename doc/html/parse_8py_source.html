116 soup = BeautifulSoup(response.text, "lxml")
117
118 house_parameters["price"] = soup.find("div", {"class": "price m-all"}).next_element.text.replace("\u2009", "")
119
120 parent_parameter_div_ResultSet = soup.find_all("div", {"class": "sc-1u2rf9n cktnNC"})
121 parameters_list = []
122
123 for child in parent_parameter_div_ResultSet:
124 parameters_list.append(child.text)
125
126 if len(parameters_list) >= 4:
127 for parameter in parameters_list:
128 if parameter.find("Площадь") != -1:
129 house_parameters["area_house"] = float(parameter.replace("Площадь", "")[0:3])
130 if parameter.find("участок") != -1:
131 house_parameters["area_land"] = float(parameter.split("участок ")[1][:-6])
132 elif parameter.find("Тип участка") != -1:
133 house_parameters["type"] = parameter.replace("Тип участка", "")
134 elif parameter.find("Дом") != -1:
135 house_parameters["house_material"] = parameter.replace("Дом", "")
136 elif parameter.find("Этажность") != -1:
137 house_parameters["num_of_floors"] = int(parameter.replace("Этажность", "")[0])
138
139 for value in house_parameters.values():
140 if value == "" or value == 0.0 or value == 0:
141 return 0
